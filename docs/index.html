RecipeJSON

RecipeJSON is a format for encoding recipes

{
  "version": 1,
  "language": "de", // bcp47
  "source": "URL",
  "author": "",
  "creationDate": "2016-11-29T12:47:03+01:00", // ISO 8601
  "modificationDate": "2016-11-29T12:47:03+01:00", // ISO 8601
  "servings": {
    "type": "persons | items",
    "count": 1
  },
  "title": "Some cool title",
  "subtitle": "Some cool subtitle", // optional
  "images": ["imageUrl"],
  "links": ["URL"], // Optional
  "tags": ["foo", "bar"],
  "recipes" [
    {
      "title": "Some cool title",
      "subtitle": "Some cool subtitle", // optional
      "images": ["imageUrl"], // Optional
      "ingredient": [
        {
          "unit": "liter",
          "amount": 1,
          "name": "Foo bar",
          "attributes": [""] // ganz
        }
      ],
      "instructions": [
        {
          "time": "P3Y6M4DT12H30M5S", // optional ISO 8601
          "text": "Waschen Sie den Weizen",
          "images": ["imageUrl"], // Optional
          "ingredients": [],
          "result": "", // optional
          "amount": 1,
          "name": "Foo bar"
        }
      ]
    }
  ]
}
